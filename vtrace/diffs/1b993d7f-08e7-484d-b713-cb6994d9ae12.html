fn find_next_impl(
    &mut self,
    len: usize,
    find_unmap_subtree: bool,
    split_huge: bool,
) -> Option<Vaddr> {
    assert_eq!(len % C::BASE_PAGE_SIZE, 0);
    let end = self.va + len;
    assert!(end <= self.barrier_va.end);
    let rcu_guard = self.rcu_guard;
    while self.va < end {
        let cur_va = self.va;
        let cur_va_range = self.cur_va_range();
        let cur_entry_fits_range = cur_va == cur_va_range.start
            && cur_va_range.end <= end;
        let mut cur_entry = self.cur_entry();
        match cur_entry.to_ref() {
            ChildRef::PageTable(pt) => {
                if find_unmap_subtree && cur_entry_fits_range
                    && (C::TOP_LEVEL_CAN_UNMAP || self.level != C::NR_LEVELS)
                {
                    return Some(cur_va);
                }
                let pt_guard = unsafe { pt.make_guard_unchecked(rcu_guard) };
                if pt_guard.nr_children() != 0 {
                    self.push_level(pt_guard);
                } else {
                    let _ = ManuallyDrop::new(pt_guard);
                    self.move_forward();
                }
                continue;
            }
            ChildRef::None => {
                self.move_forward();
                continue;
            }
            ChildRef::Frame(_, _, _) => {
                if cur_entry_fits_range || !split_huge {
                    return Some(cur_va);
                }
                let split_child = cur_entry
                    .split_if_mapped_huge(rcu_guard)
                    .expect("The entry must be a huge page");
                self.push_level(split_child);
                continue;
            }
        }
    }
    None
}

fn find_next_impl(
    &mut self,
    len: usize,
    find_unmap_subtree: bool,
    split_huge: bool,
) -> Option<Vaddr> {
    assert_eq!(len % PAGE_SIZE, 0);
    let end = self.va + len;
    assert!(end <= self.barrier_va.end);
    let end = self.va + len;
    let rcu_guard = self.rcu_guard;
    while self.va < end {
        let cur_va = self.va;
        let cur_va_range = self.cur_va_range();
        let cur_entry_fits_range = cur_va == cur_va_range.start
            && cur_va_range.end <= end;
        let mut cur_entry = self.cur_entry();
        let cur_child = cur_entry.to_ref();
        match cur_child {
            ChildRef::PageTable(pt) => {
                if find_unmap_subtree && cur_entry_fits_range
                    && (C::TOP_LEVEL_CAN_UNMAP() || self.level != C::NR_LEVELS())
                {
                    return Some(cur_va);
                }
                let mut pt_guard = pt.make_guard_unchecked(rcu_guard);
                let nr_children = pt_guard.nr_children();
                if (nr_children != 0) {
                    self.push_level(pt_guard);
                } else {
                    let _ = ManuallyDrop::new(pt_guard, Tracked(guards));
                    self.move_forward();
                }
                continue;
            }
            ChildRef::None => {
                self.move_forward();
                continue;
            }
            ChildRef::Frame(_, _, _) => {
                if cur_entry_fits_range || !split_huge {
                    return Some(cur_va);
                }
                let split_child = (cur_entry.split_if_mapped_huge(rcu_guard))
                    .expect("The entry must be a huge page");
                self.push_level(split_child);
                continue;
            }
        }
    }
    None
}
