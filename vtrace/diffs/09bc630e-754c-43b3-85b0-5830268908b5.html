pub fn insert_before(&mut self, mut frame: UniqueFrame<Link<M>>) {
    let frame_ptr = NonNull::from(frame.meta_mut());
    if let Some(current) = &mut self.current {
        let current_mut = unsafe { current.as_mut() };
        if let Some(prev) = &mut current_mut.prev {
            let prev_mut = unsafe { prev.as_mut() };
            prev_mut.next = Some(frame_ptr);
            frame.meta_mut().prev = Some(*prev);
            frame.meta_mut().next = Some(*current);
            *prev = frame_ptr;
        } else {
            frame.meta_mut().next = Some(*current);
            current_mut.prev = Some(frame_ptr);
            self.list.front = Some(frame_ptr);
        }
    } else {
        if let Some(back) = &mut self.list.back {
            unsafe {
                back.as_mut().next = Some(frame_ptr);
            }
            frame.meta_mut().prev = Some(*back);
            self.list.back = Some(frame_ptr);
        } else {
            self.list.front = Some(frame_ptr);
            self.list.back = Some(frame_ptr);
        }
    }
    frame.slot().in_list.store(self.list.lazy_get_id());
    let _ = frame.into_raw();
    self.list.size += 1;
}

pub fn insert_before(&mut self, mut frame: UniqueFrame<Link<M>>) {
    let frame_ptr = frame.meta_mut();
    let frame_ptr_as_link = MetadataAsLink::cast_from_metadata(frame_ptr);
    if let Some(current) = self.current {
        let current_md = MetadataAsLink::cast_to_metadata(current);
        let opt_prev_link: Option<ReprPtr<MetaSlot, MetadataAsLink<M>>> = borrow_field!(
            current_md => metadata.prev, & cur_perm
        );
        if let Some(prev_link) = opt_prev_link {
            let prev = MetadataAsLink::cast_to_metadata(prev_link);
            prev.metadata.next = Some(frame_ptr_as_link);
            frame_ptr.metadata.prev = Some(prev_link);
            frame_ptr.metadata.next = Some(current);
            current_md.metadata.prev = Some(frame_ptr_as_link);
        } else {
            frame_ptr.metadata.next = Some(current);
            current_md.metadata.prev = Some(frame_ptr_as_link);
            self.list.front = Some(frame_ptr_as_link);
        }
    } else {
        if let Some(back) = self.list.back {
            let back_md = MetadataAsLink::cast_to_metadata(back);
            back_md.metadata.next = Some(frame_ptr_as_link);
            frame_ptr.metadata.prev = Some(back);
            self.list.back = Some(frame_ptr_as_link);
        } else {
            self.list.front = Some(frame_ptr_as_link);
            self.list.back = Some(frame_ptr_as_link);
        }
    }
    let list_id = self.list.lazy_get_id();
    let slot = frame.slot();
    slot.in_list.store(Tracked(&mut frame_own.meta_perm.inner_perms.in_list), list_id);
    let _ = frame.into_raw();
    self.list.size += 1;
}
