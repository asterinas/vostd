pub fn take_current(&mut self) -> Option<UniqueFrame<Link<M>>> {
    let current = self.current?;
    let mut frame = {
        let meta_ptr = current.as_ptr() as *mut MetaSlot;
        let paddr = mapping::meta_to_frame::<PagingConsts>(meta_ptr as Vaddr);
        unsafe { UniqueFrame::<Link<M>>::from_raw(paddr) }
    };
    let next_ptr = frame.meta().next;
    if let Some(prev) = &mut frame.meta_mut().prev {
        let prev_mut = unsafe { prev.as_mut() };
        prev_mut.next = next_ptr;
    } else {
        self.list.front = next_ptr;
    }
    let prev_ptr = frame.meta().prev;
    if let Some(next) = &mut frame.meta_mut().next {
        let next_mut = unsafe { next.as_mut() };
        next_mut.prev = prev_ptr;
        self.current = Some(NonNull::from(next_mut));
    } else {
        self.list.back = prev_ptr;
        self.current = None;
    }
    frame.meta_mut().next = None;
    frame.meta_mut().prev = None;
    frame.slot().in_list.store(0);
    self.list.size -= 1;
    Some(frame)
}

pub fn take_current(
    &mut self,
) -> Option<(UniqueFrame<Link<M>>, Tracked<UniqueFrameOwner<Link<M>>>)> {
    let current = self.current?;
    let meta_ptr = current.addr();
    let paddr = meta_to_frame(meta_ptr);
    let (frame, frame_own) = UniqueFrame::<Link<M>>::from_raw(paddr);
    let mut frame = frame;
    let next_ptr = frame.meta().next;
    let frame_meta = frame.meta_mut();
    let opt_prev_link = borrow_field!(
        frame_meta => metadata.prev, & frame_own.meta_perm
    );
    if let Some(prev_link) = opt_prev_link {
        let prev = MetadataAsLink::cast_to_metadata(prev_link);
        prev.metadata.next = next_ptr;
    } else {
        self.list.front = next_ptr;
    }
    let prev_ptr = frame.meta().prev;
    let frame_meta = frame.meta_mut();
    let opt_next_link = frame_meta.borrow(Tracked(&frame_own.meta_perm)).metadata.next;
    if let Some(next_link) = opt_next_link {
        let next = MetadataAsLink::cast_to_metadata(next_link);
        next.metadata.prev = prev_ptr;
        self.current = Some(next_link);
    } else {
        self.list.back = prev_ptr;
        self.current = None;
    }
    let frame_meta = frame.meta_mut();
    frame_meta.metadata.next = None;
    frame_meta.metadata.prev = None;
    let slot = frame.slot();
    slot.in_list.store(Tracked(&mut inner_perms.in_list), 0);
    self.list.size -= 1;
    Some((frame, Tracked(frame_own)))
}
