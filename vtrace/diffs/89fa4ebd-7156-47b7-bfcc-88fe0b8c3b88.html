fn base_protect_query() {

    let _v1 = 1..1000;
    let _v2 = PAGE_SIZE * _v1.start..PAGE_SIZE * _v1.end;
    let _v3 = 0..PAGE_SIZE * 999;
    let _v4 = PageProperty::new_user(PageFlags::RW, CachePolicy::Writeback);
    map_untracked(&_v0, _v2.start, _v3.clone(), _v4);
    for _v5 in _v1.clone() {
        let _v6 = PAGE_SIZE * _v5;
        let (_, _v7) = _v0.page_walk(_v6).expect("Mapping should exist");
        assert_eq!(prop.flags, PageFlags::RW);
        assert_eq!(prop.cache, CachePolicy::Writeback);
    }
    let _v8 = (PAGE_SIZE * 18)..(PAGE_SIZE * 20);
    protect_range(&_v0, &_v8, |_v9| { _v9.flags -= PageFlags::W });
    for _v10 in 18..20 {
        let _v11 = PAGE_SIZE * _v10;
        let (_, _v12) = _v0.page_walk(_v11).expect("Mapping should exist");
        assert_eq!(prop.flags, PageFlags::R);
        assert_eq!(prop.cache, CachePolicy::Writeback);
    }
    let (_, _v13) = _v0.page_walk(PAGE_SIZE * 17).unwrap();
    assert_eq!(prop_before.flags, PageFlags::RW);
    let (_, _v14) = _v0.page_walk(PAGE_SIZE * 20).unwrap();
    assert_eq!(prop_after.flags, PageFlags::RW);
